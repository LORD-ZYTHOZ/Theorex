**src/flash/flush.ts:**

```typescript
// flash/flush.ts — Flush flash buffer to short-term storage.
// FLH-04: filter events >= 0.5 significance; FLH-05: clear flash after flush.
// HKS-02: called by SessionEnd hook via flash-flush CLI subcommand.

import { readFlash, writeFlash } from "./store.ts";
import { appendEntry } from "../short-term/store.ts";
import type { ShortTermEntry } from "../short-term/store.ts";

const SIGNIFICANCE_THRESHOLD = 0.5; // FLH-04

export async function flushFlash(
  sessionId: string,
  options?: {
    appendEntry?: typeof appendEntry;
    readFlash?: typeof readFlash;
    writeFlash?: typeof writeFlash;
  }
): Promise<number> {
  const _readFlash = options?.readFlash ?? readFlash;
  const _writeFlash = options?.writeFlash ?? writeFlash;
  const _appendEntry = options?.appendEntry ?? appendEntry;

  const buffer = await _readFlash(sessionId);
  const significant = buffer.events.filter(
    (e) => e.significance_score >= SIGNIFICANCE_THRESHOLD
  );

  // Write significant events to short-term (FLH-04)
  for (const event of significant) {
    const entry: ShortTermEntry = {
      id: crypto.randomUUID(),
      concept_id: 0, // flash events are tool-use records, no concept_id
      surface_form: event.tool_name,
      composite_score: event.significance_score,
      source_weight: 1.0, // Claude-generated
      timestamp: event.timestamp,
      date: event.timestamp.slice(0, 10),
    };
    await _appendEntry(entry);
  }

  // Clear flash buffer after flush (FLH-05)
  await _writeFlash({ session_id: sessionId, events: [] });

  return significant.length;
}
```

```typescript
// flash/inject.ts — Inject ACTIVE-tier context into Claude's conversation at SessionStart.
// HKS-03: reads all three lobes; prints formatted context to stdout.
// Cold start (all lobes empty): exits cleanly with empty output.

import { AxonStore } from "../axon/store.ts";
import { readShortTermFiles } from "../short-term/store.ts";
import type { ShortTermEntry } from "../short-term/store.ts";

const AXON_PATH = "data/axon.json";

export async function injectContext(
  sessionId: string,
  options?: {
    loadAxon?: () => Promise<AxonStore>;
    readShortTermFiles?: () => Promise<ShortTermEntry[]>;
  }
): Promise<string> {
  const _loadAxon = options?.loadAxon ?? (() => AxonStore.load(AXON_PATH));
  const _readShortTermFiles = options?.readShortTermFiles ?? readShortTermFiles;

  const lines: string[] = [];

  // 1. ACTIVE-tier axon nodes
  try {
    const axon = await _loadAxon();
    const activeNodes = axon.graph.nodes()
      .map((key) => axon.graph.getNodeAttributes(key))
      .filter((attrs) => attrs.relevance_tier === "ACTIVE");

    if (activeNodes.length > 0) {
      lines.push("=== THEOREX ACTIVE CONTEXT ===");
      for (const node of activeNodes.slice(0, 10)) {
        lines.push(
          `${node.surface_form} [${node.relevance_tier}/${node.sentiment_tier}]`
        );
      }
    }
  } catch {
    // Cold start or missing axon.json — no output (never throw)
  }

  // 2. Recent short-term entries (last 5 by timestamp)
  try {
    const entries = await _readShortTermFiles();
    if (entries.length > 0) {
      const recent = entries
        .slice()
        .sort((a, b) => b.timestamp.localeCompare(a.timestamp))
        .slice(0, 5);
      if (lines.length === 0) lines.push("=== THEOREX ACTIVE CONTEXT ===");
      lines.push("--- Recent short-term ---");
      for (const entry of recent) {
        lines.push(`${entry.surface_form} (score: ${entry.composite_score.toFixed(2)})`);
      }
    }
  } catch {
    // No short-term data — no output
  }

  return lines.join("\n");
}
```
