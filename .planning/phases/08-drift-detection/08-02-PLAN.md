```typescript
// NOTE: AuditEvent type must be defined locally in scorer.ts (same definition as logger.ts)
// because Plan 01 and Plan 02 run in parallel. Once both plans complete, Plan 04 CLI will
// import from logger.ts. For now, scorer.ts defines its own minimal AuditEvent copy
// (just the fields it needs: type, timestamp, concept_id, surface_form, from, to).
//
// ALTERNATIVE (if Plan 01 completes first): import { AuditEvent } from "./logger"
// Executor: use whichever approach works; keep scorer.ts free of I/O.

export interface InstabilityFlag {
  readonly concept_id: number;
  readonly surface_form: string;
  readonly dropped_at: string;   // ISO 8601 of the tier_change event
  readonly from: "ACTIVE";
  readonly to: "MILD" | "LESS";
}

export interface SentimentFlipFlag {
  readonly concept_id: number;
  readonly surface_form: string;
  readonly sentiments_seen: readonly string[];  // both PREFERRED and DISPREFERRED found in window
}

export type DriftTrend = "stable" | "drifting" | "recovering";

export function computeDriftScore(
  momentConceptIds: ReadonlySet<number>,
  activeConceptIds: ReadonlySet<number>,
): number;
// Jaccard: |intersection| / |union|
// Edge cases: momentConceptIds.size === 0 → return 1.0; activeConceptIds.size === 0 → return 0.0

export function detectInstability(
  events: readonly AuditEvent[],
  windowDays: number,
  nowMs?: number,  // defaults to Date.now()
): InstabilityFlag[];
// Returns tier_change events where: event.type==="tier_change" AND event.from==="ACTIVE"
// AND event.to !== "ACTIVE" AND event timestamp >= cutoff (nowMs - windowDays * 86_400_000)

export function detectSentimentFlips(
  events: readonly AuditEvent[],
  windowDays: number,
  nowMs?: number,  // defaults to Date.now()
): SentimentFlipFlag[];
// Returns concepts that appear with BOTH "PREFERRED" and "DISPREFERRED" within the window
// A concept with only PREFERRED or only DISPREFERRED is NOT flagged

export function classifyTrend(
  currentScore: number,
  instabilityCount: number,
): DriftTrend;
// recovering: score >= 0.7 AND instabilityCount > 0
// drifting: score < 0.5 OR instabilityCount >= 3
// stable: everything else (score >= 0.5 with instabilityCount < 3)
// Note: recovering takes precedence over stable (check recovering first)
```
